load <- function(pkg){
  new.pkg <- pkg[!(pkg %in% installed.packages()[, "Package"])]
  if (length(new.pkg))
    install.packages(new.pkg, dependencies = TRUE)
  sapply(pkg, require, character.only = TRUE)
}

N <- 5000

dataset <- c(sampletwitter, sampleblogs, samplenews)
#en_wordlist <- read.csv(paste0(getwd(),'/quiz2/en_wordlist.csv')) # downloaded fromhttp://www-personal.umich.edu/~jlawler/wordlist
obscene_words <- read.csv('/Users/pacha/Documents/Coursera/tareas\ y\ controles/Capstone\ Project/quiz2/obscene_words.csv') # downloaded from http://www.bannedwordlist.com/
stop_words <- read.csv('/Users/pacha/Documents/Coursera/tareas\ y\ controles/Capstone\ Project/quiz2/stop_words.csv') # downloaded from http://www.ranks.nl/stopwords

dataset_corpus <- Corpus(VectorSource(list(dataset)))
dataset_corpus <- tm_map(dataset_corpus, content_transformer(tolower)) #put all characters in lowercase
dataset_corpus <- tm_map(dataset_corpus, content_transformer(removePunctuation)) #remove punctuation
dataset_corpus <- tm_map(dataset_corpus, content_transformer(removeNumbers)) #remove numbers
dataset_corpus <- tm_map(dataset_corpus, removeWords, obscene_words[,1]) # remove obsecene words
dataset_corpus <- tm_map(dataset_corpus, removeWords, stop_words[,1]) # remove stop words (a, an, and, etc)
#dataset_corpus <- unlist(dataset_corpus)[!(unlist(dataset_corpus) %in% en_wordlist)] #remove mispelled words

Analysis

uniTokenizer <- function(x) NGramTokenizer(x, Weka_control(min=1, max=1))
unigrams <- DocumentTermMatrix(dataset_corpus, control=list(tokenize=uniTokenizer))
unigramDF <- data.frame(Term=unigrams$dimnames$Terms, Freq=unigrams$v)
unigramDF <- unigramDF[with(unigramDF, order(-Freq)), ]
ggplot(head(unigramDF, 10), aes(reorder(Term, -Freq), Freq)) +
  geom_bar(stat="identity") +
  ggtitle("Top 10 Unigrams by Frequency") + 
  xlab("Unigram") + ylab("Freq")
